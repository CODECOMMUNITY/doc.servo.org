Wrapper for a Reader, providing convenience methods to read the stream bit-by-bit.

Examples

extern crate compression;

use compression::bitreader::BitReader; use std::fs::File;

let f = try!(File::open("filename")); let mut br = BitReader::new(f); let byte: u8 = br.read_u8().unwrap();
